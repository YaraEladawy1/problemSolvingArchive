    if (mark > 50)
        return enMark::pass;
    else

        cout << "u failed";

{
    cout << "enter number1" << endl;
    cin >> num1;

    cout << "enter num2" << endl;
    cin >> num2;
}

int swap(int &num1, int &num2)
{
    int temp;
    temp = num1;
    num1 = num2;
    num2 = temp;
}

void PrintNum(int num1, int num2)
{
    cout << "number1 is" << num1 << endl;

    cout << "number2 is" << num2 << endl;
}

int main()
{
    int num1, num2;
    ReadNum(num1, num2);
    PrintNum(num1, num2);
    swap(num1, num2);
    PrintNum(num1, num2);
}
